    <script>
        // --- Setup ---
        const scene = new THREE.Scene();
        const camera = new THREE.PerspectiveCamera(75, window.innerWidth / window.innerHeight, 0.1, 1000);
        const renderer = new THREE.WebGLRenderer({ antialias: true, alpha: true });
        renderer.setSize(window.innerWidth, window.innerHeight);
        renderer.setPixelRatio(window.devicePixelRatio);
        document.body.appendChild(renderer.domElement);

        const labelsContainer = document.getElementById('labels');

        // --- Configuration ---
        const techStack = [
            { name: 'React', color: 0x61DAFB, distance: 10, size: 0.6, speed: 0.02 },
            { name: 'TypeScript', color: 0x3178C6, distance: 14, size: 0.5, speed: 0.015 },
            { name: 'Python', color: 0x3776AB, distance: 19, size: 0.8, speed: 0.01 },
            { name: 'PostgreSQL', color: 0x336791, distance: 25, size: 0.7, speed: 0.008 },
            { name: 'AWS', color: 0xFF9900, distance: 32, size: 1.1, speed: 0.005 },
            { name: 'Docker', color: 0x2496ED, distance: 38, size: 0.9, speed: 0.004 },
        ];

        // --- Objects ---
        
        // Starfield
        const starGeometry = new THREE.BufferGeometry();
        const starVertices = [];
        for (let i = 0; i < 10000; i++) {
            const x = (Math.random() - 0.5) * 2000;
            const y = (Math.random() - 0.5) * 2000;
            const z = (Math.random() - 0.5) * 2000;
            starVertices.push(x, y, z);
        }
        starGeometry.setAttribute('position', new THREE.Float32BufferAttribute(starVertices, 3));
        const starMaterial = new THREE.PointsMaterial({ color: 0xffffff, size: 0.7 });
        const stars = new THREE.Points(starGeometry, starMaterial);
        scene.add(stars);

        // Core (Sun)
        const sunGeometry = new THREE.SphereGeometry(4, 64, 64);
        const sunMaterial = new THREE.MeshStandardMaterial({
            color: 0x00ffff,
            emissive: 0x00ffff,
            emissiveIntensity: 1.5,
            transparent: true,
            opacity: 0.9
        });
        const sun = new THREE.Mesh(sunGeometry, sunMaterial);
        scene.add(sun);

        // Sun Glow Layer
        const glowGeometry = new THREE.SphereGeometry(4.5, 32, 32);
        const glowMaterial = new THREE.MeshBasicMaterial({ color: 0x00ffff, transparent: true, opacity: 0.1 });
        const sunGlow = new THREE.Mesh(glowGeometry, glowMaterial);
        scene.add(sunGlow);

        // Lights
        const pointLight = new THREE.PointLight(0x00ffff, 2, 200);
        scene.add(pointLight);
        const ambientLight = new THREE.AmbientLight(0xffffff, 0.1);
        scene.add(ambientLight);

        // Planets
        const planets = techStack.map(data => {
            const group = new THREE.Group();
            
            // Orbit Ring
            const ringGeom = new THREE.RingGeometry(data.distance - 0.1, data.distance + 0.1, 128);
            const ringMat = new THREE.MeshBasicMaterial({ color: 0xffffff, transparent: true, opacity: 0.1, side: THREE.DoubleSide });
            const ring = new THREE.Mesh(ringGeom, ringMat);
            ring.rotation.x = Math.PI / 2;
            scene.add(ring);

            // Planet Mesh
            const pGeom = new THREE.SphereGeometry(data.size, 32, 32);
            const pMat = new THREE.MeshStandardMaterial({ 
                color: data.color, 
                metalness: 0.9, 
                roughness: 0.1,
                emissive: data.color,
                emissiveIntensity: 0.3
            });
            const mesh = new THREE.Mesh(pGeom, pMat);
            
            // Label Element
            const label = document.createElement('div');
            label.className = 'planet-label';
            label.innerText = data.name;
            labelsContainer.appendChild(label);

            const planetObj = {
                mesh,
                data,
                angle: Math.random() * Math.PI * 2,
                label
            };
            
            scene.add(mesh);
            return planetObj;
        });

        // --- Interaction State ---
        let mouseX = 0, mouseY = 0;
        let targetRotationX = 0.5, targetRotationY = 0.5;
        let isMouseDown = false;
        let zoom = 60;

        camera.position.z = zoom;
        camera.position.y = 30;
        camera.lookAt(0, 0, 0);

        // --- Animation Loop ---
        function animate() {
            requestAnimationFrame(animate);
            const time = Date.now() * 0.001;

            // Sun pulse
            const s = 1 + Math.sin(time * 2) * 0.05;
            sun.scale.set(s, s, s);
            sunGlow.scale.set(1.2 + Math.sin(time * 2) * 0.1, 1.2 + Math.sin(time * 2) * 0.1, 1.2 + Math.sin(time * 2) * 0.1);

            // Update Planets
            planets.forEach(p => {
                p.angle += p.data.speed;
                p.mesh.position.x = Math.cos(p.angle) * p.data.distance;
                p.mesh.position.z = Math.sin(p.angle) * p.data.distance;
                p.mesh.rotation.y += 0.02;

                // Project 3D position to 2D screen for labels
                const vector = p.mesh.position.clone();
                vector.project(camera);

                const x = (vector.x * .5 + .5) * window.innerWidth;
                const y = (vector.y * -.5 + .5) * window.innerHeight;

                p.label.style.left = `${x}px`;
                p.label.style.top = `${y}px`;
                
                // Fade out labels if behind the sun or too far
                const distanceToCamera = camera.position.distanceTo(p.mesh.position);
                p.label.style.opacity = vector.z > 1 ? 0 : 1;
            });

            // Camera Smoothing
            camera.position.x += (targetRotationX * 50 - camera.position.x) * 0.05;
            camera.position.y += (targetRotationY * 50 - camera.position.y) * 0.05;
            camera.position.z += (zoom - camera.position.z) * 0.05;
            camera.lookAt(0, 0, 0);

            renderer.render(scene, camera);
        }

        // --- Events ---
        window.addEventListener('mousedown', () => isMouseDown = true);
        window.addEventListener('mouseup', () => isMouseDown = false);
        window.addEventListener('mousemove', (e) => {
            if (isMouseDown) {
                targetRotationX = (e.clientX / window.innerWidth) - 0.5;
                targetRotationY = (e.clientY / window.innerHeight) - 0.5;
            }
        });

        window.addEventListener('wheel', (e) => {
            zoom += e.deltaY * 0.05;
            zoom = Math.max(20, Math.min(100, zoom));
        });

        window.addEventListener('resize', () => {
            camera.aspect = window.innerWidth / window.innerHeight;
            camera.updateProjectionMatrix();
            renderer.setSize(window.innerWidth, window.innerHeight);
        });

        // Initialize
        window.onload = animate;

    </script>
